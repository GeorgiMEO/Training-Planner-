import tkinter as tk
from tkinter import ttk, messagebox
import json
from datetime import datetime

DATA_FILE = 'data.json'

def load_data():
    try:
        with open(DATA_FILE, 'r', encoding='utf-8') as f:
            return json.load(f)
    except (FileNotFoundError, json.JSONDecodeError):
        return []

def save_data(data):
    with open(DATA_FILE, 'w', encoding='utf-8') as f:
        json.dump(data, f, ensure_ascii=False, indent=4)

def validate_input(date_str, duration_str):
    try:
        datetime.strptime(date_str, '%Y-%m-%d')
    except ValueError:
        return False, "Дата должна быть в формате ГГГГ-ММ-ДД"
    try:
        duration = float(duration_str)
        if duration <= 0:
            raise ValueError
    except ValueError:
        return False, "Длительность должна быть положительным числом"
    return True, ""

def add_training():
    date = entry_date.get()
    training_type = entry_type.get()
    duration = entry_duration.get()

    is_valid, error_msg = validate_input(date, duration)
    if not is_valid:
        messagebox.showerror("Ошибка ввода", error_msg)
        return

    new_record = {
        "date": date,
        "type": training_type,
        "duration": float(duration)
    }

    data.append(new_record)
    save_data(data)
    update_table()

    # Очистка полей
    entry_date.delete(0, tk.END)
    entry_type.delete(0, tk.END)
    entry_duration.delete(0, tk.END)

def filter_data():
    selected_type = combo_type_filter.get()
    selected_date = entry_date_filter.get()

    filtered = data

    if selected_type and selected_type != "Все":
        filtered = [x for x in filtered if x["type"] == selected_type]

    if selected_date:
        try:
            datetime.strptime(selected_date, '%Y-%m-%d')
            filtered = [x for x in filtered if x["date"] == selected_date]
        except ValueError:
            messagebox.showerror("Ошибка", "Дата фильтра в неверном формате")
            return

    update_table(filtered)

def update_table(records=None):
    for i in tree.get_children():
        tree.delete(i)
    if records is None:
        records_to_show = data
    else:
        records_to_show = records
    for rec in records_to_show:
        tree.insert("", tk.END, values=(rec["date"], rec["type"], rec["duration"]))

# Загрузка данных при старте
data = load_data()

# Создание окна
root = tk.Tk()
root.title("Training Planner")
root.geometry("700x500")

# --- Блок добавления тренировки ---
frame_add = tk.LabelFrame(root, text="Добавить тренировку", padx=10, pady=10)
frame_add.pack(padx=10, pady=10, fill="x")

tk.Label(frame_add, text="Дата (ГГГГ-ММ-ДД):").grid(row=0, column=0, sticky="e")
entry_date = tk.Entry(frame_add, width=20)
entry_date.grid(row=0, column=1, padx=5, pady=5)

tk.Label(frame_add, text="Тип тренировки:").grid(row=1, column=0, sticky="e")
entry_type = tk.Entry(frame_add, width=20)
entry_type.grid(row=1, column=1, padx=5, pady=5)

tk.Label(frame_add, text="Длительность (мин):").grid(row=2, column=0, sticky="e")
entry_duration = tk.Entry(frame_add, width=20)
entry_duration.grid(row=2, column=1, padx=5, pady=5)

btn_add = tk.Button(frame_add, text="Добавить тренировку", command=add_training)
btn_add.grid(row=3, column=0, columnspan=2, pady=10)

# --- Блок фильтрации ---
frame_filter = tk.LabelFrame(root, text="Фильтр", padx=10, pady=10)
frame_filter.pack(padx=10, pady=10, fill="x")

types = list({x["type"] for x in data} | {"Все"})
combo_type_filter = ttk.Combobox(frame_filter, values=types, state="readonly")
combo_type_filter.set("Все")
combo_type_filter.grid(row=0, column=0, padx=5, pady=5)

tk.Label(frame_filter, text="Дата:").grid(row=0, column=1, padx=(20, 5), pady=5)
entry_date_filter = tk.Entry(frame_filter, width=20)
entry_date_filter.grid(row=0, column=2, padx=5, pady=5)

btn_filter = tk.Button(frame_filter, text="Применить фильтр", command=filter_data)
btn_filter.grid(row=0, column=3, padx=5, pady=5)

# --- Таблица тренировок ---
columns = ("Дата", "Тип", "Длительность (мин)")
tree = ttk.Treeview(root, columns=columns, show="headings")
for col in columns:
    tree.heading(col, text=col)
    tree.column(col, width=200)
tree.pack(padx=10, pady=10, fill="both", expand=True)

# Обновление таблицы при запуске
update_table()

root.mainloop()
